14. 81년도에 입사한 사원중 가장 월급 적게 받는 사람 3명만 출력하시오 (사번 , 이름  급여 , 입사일 ) 
-- 1. 별로 좋은 방법같지는 않다. 나라마다 년월일 표기 방법이 다를 수도 있다.
SELECT EMPNO, ENAME, SAL, HIREDATE
FROM (SELECT EMPNO, ENAME, SAL, HIREDATE
	FROM EMP
	WHERE SUBSTR(HIREDATE, 1, 2) = 81
	ORDER BY SAL ASC)
WHERE ROWNUM <= 3;

-- 2. TO_CHAR로 'YY' 년도 끝자리 두자리만 빼서 비교하였다.
SELECT EMPNO, ENAME, SAL, HIREDATE
FROM (SELECT EMPNO, ENAME, SAL, HIREDATE
	FROM EMP
	WHERE '81' IN TO_CHAR(HIREDATE, 'YY')
	ORDER BY SAL ASC)
WHERE ROWNUM <= 3;


15. 다음 제약을 포함한 DEPT10 테이블을 생성 
      (컬럼명과 데이터 타입은 DEPT 테이블참고) 
    DEPTNO <= PK 제약  제약명 : DEPT10_DEPTNO_PK 
    DNAME <= UK 제약  제약명 : DEPT10_DNAME_UK

CREATE TABLE DEPT10(
	DEPTNO NUMBER(4) CONSTRAINT DEPT10_DEPTNO_PK PRIMARY KEY,
	DNAME VARCHAR2(20) CONSTRAINT DEPT10_DNAME_UK UNIQUE );

# 여러 키를 묶을때 이렇게 묶고 ()안에 들어가는 컬럼명을 복수로 써도 된다.
CREATE TABLE DEPT10(
	DEPTNO NUMBER(4),
	DNAME VARCHAR2(20),
	CONSTRAINT DEPT10_DEPTNO_PK PRIMARY KEY(DEPTNO),
	CONSTRAINT DEPT10_DNAME_UK UNIQUE KEY(DNAME) NOT NULL);

(16~18) EMP10 테이블에 여러가지 제약을 여러가지 제약을 추가해보자. 

16.  EMPNO <= PK : 제약명 : EMP10_EMPNO_PK 

ALTER TABLE EMP10
ADD CONSTRAINT EMP10_EMPNO_PK PRIMARY KEY (EMPNO)

17.  ENAME <= UNIQUE 제약명 : EMP10_ENAME_UK 

ALTER TABLE EMP10

18.  DEPTNO <= FK  제약명 : EMP10_DEPTNO_FK
	참조컬럼은 DEPT10 테이블의 DEPTNO 컬럼 

ALTER TABLE DEPT10
ADD CONSTRAINT EMP10_DEPTNO_FK FOREIGN KEY (DEPTNO)
REFERENCES DEPT10(DEPTNO)
