## B. 제시된 문제 풀기

STEP 1 
사원명 입사일 근속일
               일

SELECT E.ENAME, E.HIREDATE, 
	TRUNC(SYSDATE - E.HIREDATE, 0) AS "SERV_DAY"
FROM EMP E
WHERE E.ENAME = 'SCOTT'


STEP 2
사원명 입사일 근속월
                월(소숫점 버림)
SELECT E.ENAME, E.HIREDATE, TRUNC(MONTHS_BETWEEN(SYSDATE, E.HIREDATE), 0) AS "SERV_MONTH"
FROM EMP E
WHERE E.ENAME = 'SCOTT'

STEP 3
사원명 입사일 근속년 근속월
SELECT E.ENAME, E.HIREDATE, 
	TRUNC(MONTHS_BETWEEN(SYSDATE, E.HIREDATE)/12, 0) AS "SERV_YEAR",
	TRUNC(MONTHS_BETWEEN(SYSDATE, E.HIREDATE), 0) AS "SERV_MONTH"
FROM EMP E
WHERE E.ENAME = 'SCOTT'

STEP 4
사원명 입사일 근속기간
              YY년 M개월
SELECT E.ENAME, E.HIREDATE, 
		TO_CHAR(TRUNC(MONTHS_BETWEEN(SYSDATE, E.HIREDATE)/12, 0))|| 
		'년'||
		TO_CHAR(TRUNC(MOD(MONTHS_BETWEEN(SYSDATE, E.HIREDATE),12), 0))||
		'개월' AS "SERV_YY/MM"
FROM EMP E
WHERE E.ENAME = 'SCOTT'



-- 몰라도 됨
-- COL "SERV_YY/MM" format a10 으로 컬럼명의 출력열 길이 설정 가능
